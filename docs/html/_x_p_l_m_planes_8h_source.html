82 /*
83  * XPLMPlaneDrawState_t
84  *
85  * This structure contains additional plane parameter info to be passed to
86  * draw plane. Make sure to fill in the size of the structure field with
87  * sizeof(XPLMDrawPlaneState_t) so that the XPLM can tell how many fields you
88  * knew about when compiling your plugin (since more fields may be added
89  * later).
90  *
91  * Most of these fields are ratios from 0 to 1 for control input. X-Plane
92  * calculates what the actual controls look like based on the .acf file for
93  * that airplane. Note for the yoke inputs, this is what the pilot of the
94  * plane has commanded (post artificial stability system if there were one)
95  * and affects aelerons, rudder, etc. It is not necessarily related to the
96  * actual position of the plane!
97  *
98  */
99 typedef struct {
100  /* The size of the draw state struct. */
101  int structSize;
102  /* A ratio from [0..1] describing how far the landing gear is extended. */
103  float gearPosition;
104  /* Ratio of flap deployment, 0 = up, 1 = full deploy. */
105  float flapRatio;
106  /* Ratio of spoiler deployment, 0 = none, 1 = full deploy. */
107  float spoilerRatio;
108  /* Ratio of speed brake deployment, 0 = none, 1 = full deploy. */
109  float speedBrakeRatio;
110  /* Ratio of slat deployment, 0 = none, 1 = full deploy. */
111  float slatRatio;
112  /* Wing sweep ratio, 0 = forward, 1 = swept. */
113  float wingSweep;
114  /* Thrust power, 0 = none, 1 = full fwd, -1 = full reverse. */
115  float thrust;
116  /* Total pitch input for this plane. */
117  float yokePitch;
118  /* Total Heading input for this plane. */
119  float yokeHeading;
120  /* Total Roll input for this plane. */
121  float yokeRoll;
122 } XPLMPlaneDrawState_t;
